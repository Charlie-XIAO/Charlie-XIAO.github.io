Solution. We implement the bisection method as follows. 
function bisection(func, left, right, tolerance)
    """Bisection method.

    Parameters
    ----------
    func : function
        The function to find a root for.
    left : Float64, etc.
        The left boundary of the interval to find a root.
    right : Float64, etc.
        The right boundary of the interval to find a root.
    tolerance : Float64, etc.
        The tolerance for the value of the root.

    Returns
    -------
    root : Float64 or boolean
        If a root is successfully found within tolerance, then it will be returned.
        Otherwise, the function returns false.
    """
    mid = (left + right) / 2
    mid_val = func(mid)
    if (abs(mid_val) < tolerance)
        return mid
    elseif (right - left < 1e-20)
        return false
    elseif (mid_val * func(left) < 0)
        return bisection(func, left, mid, tolerance)
    else
        return bisection(func, mid, right, tolerance)
    end
end

function newton(f, f_, x0, tolerance, maxiter)
    """Newton's method.

    Parameters
    ----------
    f : function
        The function to find a root for.
    f_ : function
        The derivative of the function to find a root for.
    x0 : Float64, etc.
        The initial guess of the root, better near the real root.
    tolerance : Float64, etc.
        The tolerance for the value of the root.
    maxiter : Int64, etc.
        The maximum number of iterations to run Newton's method.

    Returns
    -------
    root : Float64
        The root found by Newton's method.
    """
    cur_x, cur_f = x0, f(x0)
    for _ = 1 : maxiter
        next_x = cur_x - cur_f / f_(cur_x)
        if (abs(next_x - cur_x) < tolerance)
            break
        end
        cur_x, cur_f = next_x, f(next_x)
    end
    return cur_x
end

function newton_hist(f, f_, x0, tolerance, maxiter)
    """Newton's method with iteration history.

    Parameters
    ----------
    f : function
        The function to find a root for.
    f_ : function
        The derivative of the function to find a root for.
    x0 : Float64, etc.
        The initial guess of the root, better near the real root.
    tolerance : Float64, etc.

    maxiter : Int64, etc.

    Returns
    -------
    x_hist : 1D array
        The history of the x values during the Newton's method.
    f_list : 1D array
        The history of the f(x) values during the Newton's method.
    """
    x_hist, f_hist = [], []
    cur_x, cur_f = x0, f(x0)
    for _ = 1 : maxiter
        next_x = cur_x - cur_f / f_(cur_x)
        if (abs(next_x - cur_x) < tolerance)
            break
        end
        push!(x_hist, cur_x)
        push!(f_hist, cur_f)
        cur_x, cur_f = next_x, f(next_x)
    end
    return x_hist, f_hist
end;

Problem 1.5 - Secant Convergence

In this problem, you will prove the rate of convergence for the secant method.
(a) Show that the secant method $$ \begin{align*} x_{k+1}=x_k-\frac{x_k-x_{k-1}}{f(x_k)-f(x_{k-1})}f(x_k) \end{align*} $$ can be rewritten in the form $$ \begin{align*} x_{k+1}=\frac{x_kf(x_{k-1})-x_{k-1}f(x_k)}{f(x_{k-1})-f(x_k)}. \end{align*} $$
Proof. We can write that $$ \begin{align*} x_{k+1}=x_k-\frac{x_k-x_{k-1}}{f(x_k)-f(x_{k-1})}f(x_k)={} & \frac{x_k(f(x_k)-f(x_{k-1}))-f(x_k)(x_k-x_{k-1})}{f(x_k)-f(x_{k-1})} \nonumber\\ ={} & \frac{x_{k-1}f(x_k)-x_kf(x_{k-1})}{f(x_k)-f(x_{k-1})}=\frac{x_kf(x_{k-1})-x_{k-1}f(x_k)}{f(x_{k-1})-f(x_k)}, \end{align*} $$ which gives the desired result.
(b) Now, denote the root of $f$ to be $\xi$, so that $f(\xi)=0$. Also assume that $f$ is twice continuously differentiable and that $f'>0$ and $f''>0$ in a neighborhoood of $\xi$. Define the quantity $\phi$ to be $$ \begin{align*} \phi(x_k,x_{k-1})=\frac{x_{k+1}-\xi}{(x_k-\xi)(x_{k-1}-\xi)}. \end{align*} $$ Compute (for fixed value of $x_{k-1}$) that $$ \begin{align*} \psi(x_{k-1})=\lim_{x_k\rightarrow\xi}\phi(x_k,x_{k-1}). \end{align*} $$
Solution. We can compute that $$ \begin{align*} \psi(x_{k-1})={} & \lim_{x_k\rightarrow\xi}\phi(x_k,x_{k-1})=\lim_{x_k\rightarrow\xi}\frac{x_{k+1}-\xi}{(x_k-\xi)(x_{k-1}-\xi)}=\lim_{x_k\rightarrow\xi}\frac{\frac{x_kf(x_{k-1})-x_{k-1}f(x_k)}{f(x_{k-1})-f(x_k)}-\xi}{(x_k-\xi)(x_{k-1}-\xi)} \nonumber\\ ={} & \lim_{x_k\rightarrow\xi}\frac{x_kf(x_{k-1})-x_{k-1}f(x_k)-\xi f(x_{k-1})+\xi f(x_k)}{(f(x_{k-1})-f(x_k))(x_k-\xi)(x_{k-1}-\xi)} \nonumber\\ ={} & \lim_{x_k\rightarrow\xi}\frac{(x_k-\xi)f(x_{k-1})-(x_{k-1}-\xi)f(x_k)}{(f(x_{k-1})-f(x_k))(x_k-\xi)(x_{k-1}-\xi)} \nonumber\\ ={} & \frac{f(x_{k-1})}{x_{k-1}-\xi}\cdot\lim_{x_k\rightarrow\xi}\frac{1}{f(x_{k-1})-f(x_k)}-\lim_{x_k\rightarrow\xi}\frac{1}{f(x_{k-1})-f(x_k)}\cdot\lim_{x_k\rightarrow\infty}\frac{f(x_k)}{x_k-\xi} \nonumber\\ ={} & \frac{f(x_{k-1})}{x_{k-1}-\xi}\cdot\frac{1}{f(x_{k-1})-f(\xi)}-\frac{1}{f(x_{k-1})-f(\xi)}\cdot\lim_{x_k\rightarrow\infty}\frac{f(x_k)-f(\xi)}{x_k-\xi} \nonumber\\ ={} & \frac{1}{x_{k-1}-\xi}-\frac{f'(\xi)}{f(x_{k-1})-f(\xi)}=\frac{f(x_{k-1})-x_{k-1}f'(\xi)+\xi f'(\xi)}{(x_{k-1}-\xi)f(x_{k-1})}. \end{align*} $$
(c) Now compute $$ \begin{align*} \lim_{x_{k-1}\rightarrow\xi}\psi(x_{k-1}), \end{align*} $$ and therefore show that $$ \begin{align*} \lim_{x_k,x_{k-1}\rightarrow\xi}\phi(x_k,x_{k-1})=\frac{f''(\xi)}{2f'(\xi)}. \end{align*} $$
Proof. We can compute that $$ \begin{align*} \lim_{x_{k-1}\rightarrow\xi}\psi(x_{k-1})={} & \lim_{x_{k-1}\rightarrow\xi}\frac{f(x_{k-1})-x_{k-1}f'(\xi)+\xi f'(\xi)}{(x_{k-1}-\xi)f(x_{k-1})}\overset{\text{L'Hosp.}}{=}\lim_{x_{k-1}\rightarrow\xi}\frac{f'(x_{k-1})-f'(\xi)}{f(x_{k-1})+(x_{k-1}-\xi)f'(x_{k-1})} \nonumber\\ \overset{\text{L'Hosp.}}{=}{} & \lim_{x_{k-1}\rightarrow\xi}\frac{f''(x_{k-1})}{f'(x_{k-1})+f'(x_{k-1})+(x_{k-1}-\xi)f''(x_{k-1})}=\frac{f''(\xi)}{2f'(\xi)}. \end{align*} $$ Therefore, we can conclude that $$ \begin{align*} \lim_{x_k,x_{k-1}\rightarrow\xi}\phi(x_k,x_{k-1})=\lim_{x_{k-1}\rightarrow\xi}\psi(x_{k-1})=\frac{f''(\xi)}{2f'(\xi)}, \end{align*} $$ and the proof is done.
(d) Next, assume that the secant method has convergence order $q$, that is to say that $$ \begin{align*} \lim_{k\rightarrow\infty}\frac{|x_{k+1}-\xi|}{|x_k-\xi|^q}=A<\infty. \end{align*} $$ Using the above results, show that $q-1-1/q=0$, and therefore that $q=(1+\sqrt{5})/2$.
Proof. Assuming that the secant method has convergence order $q$, we have that $$ \begin{align*} \lim_{k\rightarrow\infty}|x_{k+1}-\xi|=A\lim_{k\rightarrow\infty}|x_k-\xi|^q\implies\lim_{k\rightarrow\infty}\ln|x_{k+1}-\xi|=\ln A+q\lim_{k\rightarrow\infty}\ln|x_k-\xi|. \end{align*} $$ Denote $\epsilon_{t}:=\lim_{k\rightarrow\infty}\ln|x_t-\xi|$, $t\in\set{k-1,k,k+1}$, then we can rewrite and deduce that $$ \begin{align*} \epsilon_{k+1}=\ln A+q\epsilon_k,\quad\epsilon_k=\ln A+q\epsilon_{k-1}\implies\epsilon_{k-1}=\frac{\epsilon_k-\ln A}{q}. \end{align*} $$ Bringing in this result, now we can compute that $$ \begin{align*} \lim_{k\rightarrow\infty}\ln\phi(x_k,x_{k-1})=\epsilon_{k+1}-\epsilon_k-\epsilon_{k-1}=\bkt{1+\frac{1}{q}}\ln A+\bkt{q-1-\frac{1}{q}}\epsilon_k. \end{align*} $$ Note that $\epsilon_k$ is (negative) infinity, since $x_k\rightarrow\xi$ as $k\rightarrow\infty$. However, the left-hand side is finite, which can be deduced from the limit. Therefore, we must require $$ \begin{align*} q-1-\frac{1}{q}=0\implies q^2-q-1=0\implies q=\frac{1+\sqrt{5}}{2}\quad\text{(ignoring negative solution)}. \end{align*} $$ This is exactly the desired result.
(e) Finally, show that this implies that $$ \begin{align*} \lim_{k\rightarrow\infty}\frac{|x_{k+1}-\xi|}{|x_k-\xi|^q}=\bkt{\frac{f''(\xi)}{2f'(\xi)}}^{q/(1+q)}. \end{align*} $$
Proof. Using the limit and the previous result on $q$, we can rewrite that $$ \begin{align*} \ln\bkt{\frac{f''(\xi)}{2f'(\xi)}}=\bkt{1+\frac{1}{q}}\ln A\implies\ln A=\frac{q}{q+1}\cdot\ln\bkt{\frac{f''(\xi)}{2f'(\xi)}}\implies A=\bkt{\frac{f''(\xi)}{2f'(\xi)}}^{q/(1+q)}. \end{align*} $$ Therefore, we can conclude that $$ \begin{align*} \lim_{k\rightarrow\infty}\frac{|x_{k+1}-\xi|}{|x_k-\xi|^q}=A=\bkt{\frac{f''(\xi)}{2f'(\xi)}}^{q/(1+q)}, \end{align*} $$ which proves the desired result.

Problem 1.6 - Vector and Matrix Norm Inequalities

(a) Let $\mathbf{v}=(v_1,\cdots,v_n)^\top\in\RR^n$. Prove the following two inequalities: $$ \begin{align*} \norm{\mathbf{v}}_{\infty}\leq{} & \norm{\mathbf{v}}_2, \\ \norm{\mathbf{v}}_2^2\leq{} & \norm{\mathbf{v}}_1\norm{\mathbf{v}}_{\infty}. \end{align*} $$ For each inequality, given an example of a non-zero vector $\mathbf{v}$ for which equality is obtained. Finally, show that $$ \begin{align*} \norm{\mathbf{v}}_{\infty}\leq{} & \norm{\mathbf{v}}_2\leq\norm{\mathbf{v}}_1, \\ \norm{\mathbf{v}}_2\leq{} & \sqrt{n}\norm{\mathbf{v}}_{\infty}. \end{align*} $$
Proof. For the first inequality, we have that $$ \begin{align*} \norm{\mathbf{v}}_{\infty}=\max_{1\leq i\leq n}|v_i|=\max_{1\leq i\leq n}\sqrt{|v_i|^2}=\sqrt{\max_{1\leq i\leq n}|v_i|^2}\leq\sqrt{\sum_{i=1}^n|v_i|^2}=\norm{\mathbf{v}}_2, \end{align*} $$ where the equality holds when $\mathbf{v}$ has at most one non-zero component, for instance, $\mathbf{v}=(1,0,\cdots,0)^\top$. For the second inequality, we have that $$ \begin{align*} \norm{\mathbf{v}}_2^2=\sum_{i=1}^n|v_i|^2=\sum_{i=1}^n|v_i|\bkt{\max_{1\leq i\leq n}|v_i|}=\max_{1\leq i\leq n}|v_i|\cdot\sum_{i=1}^n|v_i|=\norm{\mathbf{v}}_{\infty}\norm{\mathbf{v}}_1, \end{align*} $$ where the equality holds when all components of $\mathbf{v}$ are equal, for instance, $\mathbf{v}=(1,\cdots,1)^\top$. For the third inequality, the left-hand side has already been shown, so it suffices to prove the right-hand side. Indeed, we have that $$ \begin{align*} \norm{\mathbf{v}}_2=\sqrt{\sum_{i=1}^n|v_i|^2}\leq\sqrt{\sum_{i=1}^n|v_i|^2+\sum_{i\lt j}|v_i||v_j|}=\sqrt{\bkt{\sum_{i=1}^n|v_i|}^2}=\sum_{i=1}^n|v_i|=\norm{\mathbf{v}}_1, \end{align*} $$ where the equality holds when $\mathbf{v}=\mathbf{0}$. Finally, for the last inequality, we can show that $$ \begin{align*} \norm{\mathbf{v}}_2=\sqrt{\sum_{i=1}^n|v_i|^2}\leq\sqrt{\sum_{i=1}^n\bkt{\max_{1\leq j\leq n}|v_i|}^2}=\sqrt{n\bkt{\max_{1\leq j\leq n}|v_i|}^2}=\sqrt{n}\max_{1\leq j\leq n}|v_i|=\sqrt{n}\norm{\mathbf{v}}_{\infty}, \end{align*} $$ where the equality holds when all components of $\mathbf{v}$ are equal, for instance, $\mathbf{v}=(1,\cdots,1)^\top$.
(b) Next, let $A\in\RR^{m\times n}$ be any real-valued matrix. Show that $$ \begin{align*} \norm{A}_{\infty}\leq{} & \sqrt{n}\norm{A}_2, \\ \norm{A}_2\leq{} & \sqrt{m}\norm{A}_{\infty}, \end{align*} $$ where above, the norms are interpreted as the induced matrix norms. For each inequality, give an example matrix $A$ for which equality is attained.
Proof. Problem 1.7 - Frobenius Matrix Norm

Another commonly used matrix norm in linear algebra is the Frobenius norm, defined for a matrix $A\in\RR_{m\times n}$ as $$ \begin{align*} \norm{A}_F=\sqrt{\sum_{i=1}^m\sum_{j=1}^n|a_{ij}|^2}. \end{align*} $$
(a) Prove that $\norm{A}_F=\sqrt{\tr(A^\top A)}$.
Proof. We can compute that $$ \begin{align*} A^\top A= \begin{pmatrix} a_{11} & \cdots & a_{m1} \\ \vdots & \ddots & \vdots \\ a_{1n} & \cdots & a_{mn} \end{pmatrix} \begin{pmatrix} a_{11} & \cdots & a_{1n} \\ \vdots & \ddots & \vdots \\ a_{m1} & \cdots & a_{mn} \end{pmatrix} = \begin{pmatrix} \sum_{i=1}^ma_{i1}^2 & \cdots & \sum_{i=1}^ma_{i1}a_{in} \\ \vdots & \ddots & \vdots \\ \sum_{i=1}^ma_{i1}a_{in} & \cdots & \sum_{i=1}^ma_{in}^2 \end{pmatrix}. \end{align*} $$ Therefore, the trace of $A^\top A$ can be computed as $$ \begin{align*} \tr(A^\top A)=\sum_{j=1}^n(A^\top A)_{jj}=\sum_{j=1}^n\sum_{i=1}^ma_{ij}^2=\sum_{i=1}^m\sum_{j=1}^n|a_{ij}|^2. \end{align*} $$ Therefore, by definition of the Frobenius norm, we can conclude that $$ \begin{align*} \norm{A}_F=\sqrt{\sum_{i=1}^m\sum_{j=1}^n|a_{ij}|^2}=\sqrt{\tr(A^\top A)}, \end{align*} $$ and the proof is complete.
(b) Prove also that $\norm{A}_F=\sqrt{\sum_j\sigma_j^2}$, where $\sigma_j$ are the singular values of $A$.
Proof. By singular value decomposition (for real matrices), we can write $$ \begin{align*} A=U\Sigma V^\top, \end{align*} $$ where $U\in\RR^{m\times m}$ and $V\in\RR^{n\times n}$ are unitary matrices and $\Sigma\in\RR^{m\times n}$ is a rectangular diagonal matrix with diagonal entries equal to the singular values of $A$. Therefore, we can expand $$ \begin{align*} \norm{A}_F=\sqrt{\tr(A^\top A)}=\sqrt{\tr(V\Sigma^\top U^\top U\Sigma V^\top)}=\sqrt{\tr(V\Sigma^\top\Sigma V^\top)}=\sqrt{\tr(\Sigma^\top\Sigma V^\top V)}=\sqrt{\tr(\Sigma^\top\Sigma)}, \end{align*} $$ where we make use of the property $\tr(MN)=\tr(NM)$ for any matrix $M$ and $N$, and that $M^\top M=I$ for any unitary real matrix $M$. Now note that $\Sigma^\top\Sigma$ is an $n\times n$ diagonal matrix with diagonal entries equal to the square of the singular values of $A$. Therefore, we can conclude that $$ \begin{align*} \norm{A}_F=\sqrt{\sum_{j=1}^n\sigma_j^2}, \end{align*} $$ where $\sigma_j$ are the singular values of $A$, and the proof is complete.
(c) Show that $\norm{A}_2\leq\norm{A}_F$.
Proof. For an arbitrary $\mathbf{x}\in\RR^n$, we can see that $$ \begin{align*} \norm{A\mathbf{x}}_2={} & \sqrt{\mathbf{x}^\top A^\top A\mathbf{x}}=\sqrt{\mathbf{x}^\top V\Sigma^\top U^\top U\Sigma V^\top\mathbf{x}}=\sqrt{\mathbf{x}^\top V\Sigma^\top\Sigma V^\top\mathbf{x}}=\sqrt{\tr(\mathbf{x}^\top V\Sigma^\top\Sigma V^\top\mathbf{x})} \nonumber\\ ={} & \sqrt{\tr(\mathbf{x}^\top\Sigma^\top\Sigma V^\top V\mathbf{x})}=\sqrt{\tr(\mathbf{x}^\top\Sigma^\top\Sigma\mathbf{x})}=\sqrt{\mathbf{x}^\top\Sigma^\top\Sigma\mathbf{x}}=\sqrt{\sum_{j=1}^n\sigma_j^2x_j^2}. \end{align*} $$ Also, note that $$ \begin{align*} \sum_{j=1}^n\sigma_j^2x_j^2\leq\sum_{j=1}^n\sigma_j^2x_j^2+\sum_{i\neq j}\sigma_ix_j=\bkt{\sum_{i=1}^n\sigma_i^2}\bkt{\sum_{i=1}^nx_i^2}. \end{align*} $$ Therefore, we can conclude that $$ \begin{align*} \norm{A}_2=\max_{\norm{\mathbf{x}}_2=1}\norm{A\mathbf{x}}_2={} & \max_{\norm{\mathbf{x}}_2=1}\sqrt{\sum_{j=1}^n\sigma_j^2x_j^2}\leq\max_{\norm{\mathbf{x}}_2=1}\sqrt{\bkt{\sum_{i=1}^n\sigma_i^2}\bkt{\sum_{i=1}^nx_i^2}} \nonumber\\ ={} & \max_{\norm{\mathbf{x}}_2=1}\sqrt{\sum_{i=1}^n\sigma_i^2}=\sqrt{\sum_{i=1}^n\sigma_i^2}=\norm{A}_F, \end{align*} $$ where the equality holds when all the singular values of $A$ are zero, and the proof is complete.

function classical_gs(n, A)
    """Classical Gram-Schmidt orthogonalization algorithm.

    Parameters
    ----------
    n : Int64, etc.
        The dimension of the matrix to be orthogonalized.
    A : Matrix{Float64}, etc.
        The matrix to be orthogonalized.

    Returns
    -------
    Q : Matrix{Float64}
        The orthogonalized matrix via the classical Gram-Schmidt algorithm.
    """
    Q = Array{Float64}(undef, n, n)
    for j = 1 : n
        v = copy(A[:, j])
        for i = 1 : (j - 1)
            v .-= dot(Q[:, i], A[:, j]) * Q[:, i]
        end
        Q[:, j] = v / norm(v)
    end
    return Q
end

However, as we have seen, Newton's method requires an initialization close to the root one wants to find, and it can be difficult to find all roots of a polynomial. Luckily, one can use the following relationship between eigenvalues and polynomial roots to find all roots of a given polynomial at once: Let us consider a polynomial of degree $n$ with leading coefficient $a_n=1$: $$ \begin{align*} p(x)=a_0+a_1x+\cdots+a_{n-1}x^{n-1}+x^n,\qquad\text{with $a_i\in\RR$}. \end{align*} $$
(a) Show that $p$ is the characteristic polynomial of the matrix $$ \begin{align*} A_p= \begin{pmatrix} 0 & & & -a_0 \\ 1 & \ddots & & -a_1 \\ & \ddots & 0 & \vdots \\ & & 1 & -a_{n-1} \end{pmatrix}\in\RR^{n\times n}, \end{align*} $$ which is sometimes called a companion matrix for $p$. Thus, the roots of $p$ can be computed as the eigenvalues of $A_p$ using, for example, the QR algorithm. Julia's eigen function can do this computation.
Solution. By cofactor expansion with respect to the first row, we have that $$ \begin{align*} & \det(A_p-xI)=\det \begin{pmatrix} -x & & & -a_0 \\ 1 & \ddots & & -a_1 \\ & \ddots & -x & \vdots \\ & & 1 & -a_{n-1}-x \end{pmatrix} \nonumber\\ & \qquad\qquad=-x\cdot\det \begin{pmatrix} -x & & & -a_1 \\ 1 & \ddots & & -a_2 \\ & \ddots & -x & \vdots \\ & & 1 & -a_{n-1}-x \end{pmatrix}+(-1)^{n+1}(-a_0)\cdot\det \underbrace{\begin{pmatrix} 1 & -x & & \\ 0 & \ddots & \ddots & \\ & \ddots & \ddots & -x \\ & & 0 & 1 \end{pmatrix}}_{\text{upper triangular}} \nonumber\\ & \qquad\qquad=-x\cdot\det \begin{pmatrix} -x & & & -a_1 \\ 1 & \ddots & & -a_2 \\ & \ddots & -x & \vdots \\ & & 1 & -a_{n-1}-x \end{pmatrix}+(-1)^na_0. \end{align*} $$ Similarly, we can write that $$ \begin{align*} \det \begin{pmatrix} -x & & & -a_1 \\ 1 & \ddots & & -a_2 \\ & \ddots & -x & \vdots \\ & & 1 & -a_{n-1}-x \end{pmatrix}=-x\det \begin{pmatrix} -x & & & -a_2 \\ 1 & \ddots & & -a_3 \\ & \ddots & -x & \vdots \\ & & 1 & -a_{n-1}-x \end{pmatrix}-(-1)^na_1. \end{align*} $$ Plugging this result into the previous expression, we can obtain that $$ \begin{align*} \det(A_p-xI)=x^2\det \begin{pmatrix} -x & & & -a_2 \\ 1 & \ddots & & -a_3 \\ & \ddots & -x & \vdots \\ & & 1 & -a_{n-1}-x \end{pmatrix}+(-1)^na_1x+(-1)^na_0. \end{align*} $$ Iterating this process, we can finally obtain that $$ \begin{align*} \det(A_p-xI)={} & (-x)^{n-2}\det \begin{pmatrix} -x & -a_{n-2} \\ 1 & -a_{n-1}-x \end{pmatrix}+(-1)^n(a_{n-3}x^{n-3}+\cdots+a_1x+a_0) \nonumber\\ ={} & (-x)^{n-2}(a_{n-1}x+x^2+a_{n-2})+(-1)^n(a_{n-3}x^{n-3}+\cdots+a_1x+a_0) \nonumber\\ ={} & (-1)^n(x^n+a_{n-1}x^{n-1}+a_{n-2}x^{n-2}+a_{n-3}x^{n-3}+\cdots+a_1x+a_0). \end{align*} $$ By keeping the leading coefficient as $1$, this is exactly $p(x)$. A more formal proof can be done via mathematical induction, but I think this way of writing it is more intrinsic.

function pwcoefs(n)
    """Returns the coefficients of Wilkinson's polynomial.

    Parameters
    ----------
    n : Int64, etc.
        The order of the Wilkinson's polynomial, i.e., the polynomial
        (x - 1)(x - 2)(x - 3) ... (x - n)
    
    Returns
    -------
    coeffs : Array{Float64}, etc.
        The array of coefficients, starting from the constant term.

    """
    coeffs = zeros(n + 1)
    coeffs[1] = 1
    for i in 1 : n
        for j in (i + 1) : -1 : 2
            coeffs[j] = coeffs[j - 1] - i * coeffs[j]
        end
        coeffs[1] = -i * coeffs[1]
    end
    return coeffs
end

Problem 3.3 - Matrix Functions

Let $A\in\CC^{n\times n}$ be a \textit{normal} matrix. For $t\in\RR$, the matrix exponential is defined as $$ \begin{align*} e^{tA}=I+tA+\frac{(tA)^2}{2!}+\frac{(tA)^3}{3!}+\cdots=\sum_{k=0}^{\infty}\frac{(tA)^k}{k!}. \end{align*} $$
(a) Can you relate the eigenvalue decomposition of $A$ to that of $e^{tA}$?
Solution. Suppose that $A$ can be decomposed as $$ \begin{align*} A=Q\Lambda Q^{-1}, \end{align*} $$ where $\Lambda$ is a diagonal matrix with diagonal entries the eigenvalues of $A$, and $Q$ is the matrix in which the $i$th column is the eigenvector corresponding to the $i$th eigenvalue in $\Lambda$. Then, we can compute that $$ \begin{align*} tA={} & Q\Lambda Q^{-1}, \\ (tA)^2={} & t^2Q\Lambda Q^{-1}Q\Lambda Q^{-1}=t^2Q\Lambda^2Q^{-1}, \\ \vdots \nonumber\\ (tA)^k={} & t^kQ\Lambda^kQ^{-1}. \end{align*} $$ Then, by definition of the matrix exponential, we have that $$ \begin{align*} e^{tA}=\sum_{k=0}^{\infty}\frac{t^kQ\Lambda^kQ^{-1}}{k!}=Q\bkt{\sum_{k=0}^{\infty}\frac{(t\Lambda)^k}{k!}}Q^{-1}=Qe^{t\Lambda}Q^{-1}. \end{align*} $$
(b) Now further suppose that the matrix $A$ is symmetric positive definite. If $X$ is a matrix such that $X^*X=A$, then $X$ is referred to as the square root of $A$. If you are given the eigendecomposition of $A$, i.e., $A=UDU^{-1}$, can you find the square root of $A$?
Solution. Denote the diagonal elements, i.e., the eigenvalues of $A$ as $\lambda_1,\cdots,\lambda_n$. Note that since $A$ is symmetric positive definite, all these eigenvalues are real and positive. Now consider the diagonal matrix $S$ with diagonal elements $\sqrt{\lambda_1},\cdots,\sqrt{\lambda_n}$. Then, taking $X=USU^{-1}$, we can compute that $$ \begin{align*} X^*X=\underbrace{(U^{-1})^*}_{=U}S^*\underbrace{U^*U}_{=I}SU^{-1}=US^*SU^{-1}=UDU^{-1}=A, \end{align*} $$ since $U$ is unitary due to the symmetry of $A$, and that $$ \begin{align*} S^*S= \begin{pmatrix} \sqrt{\lambda_1} & & \\ & \ddots & \\ & & \sqrt{\lambda_n} \end{pmatrix} \begin{pmatrix} \sqrt{\lambda_1} & & \\ & \ddots & \\ & & \sqrt{\lambda_n} \end{pmatrix}= \begin{pmatrix} \lambda_1 & & \\ & \ddots & \\ & & \lambda_n \end{pmatrix}=D. \end{align*} $$ Therefore, we can conclude that the square root of $A$ is $X=USU^{-1}$.
(c) Is the matrix square root unique? If not, provide an alternative square root matrix to the one in the previous part.
Solution. The matrix square root is not unique. Take $Y=SU^{-1}$, then we can check that $$ \begin{align*} Y^*Y=\underbrace{(U^{-1})^*}_{=U}S^*SU^{-1}=US^*SU^{-1}=UDU^{-1}. \end{align*} $$
(d) Motivated by the previous questions, can you determine a method for computing general matrix functions of real symmetric matrices? That is, if $f=f(x)$ is a scalar function, how can one define $f(A)$?
Solution. Using the same notations as the question above, we define $f(A)$ such that $$ \begin{align*} f(A)=U \begin{pmatrix} f(\lambda_1) & & \\ & \ddots & \\ & & f(\lambda_n) \end{pmatrix} U^{-1}. \end{align*} $$
(e) In class, we saw how to use Newton's method to find the square root of a number $a$ by finding the root of the function $f(x)=x^2-a$. Can you use a similar Newton iteration to find the square root of a symmetric positive definite matrix $A$ by finding the root of some matrix function $f=f(A)$? Implement your algorithm and use it to find the square root of the $6\times6$ matrix $A$ with entries $$ \begin{align*} A_{ij}=5\delta_{ij}+\sqrt{i^2+j^2}. \end{align*} $$
Solution. Analogous to Newton's method, we use a similar iteration such that $$ \begin{align*} X_{k+1}=X_k-(2X_k)^{-1}(X_k^2-A)=X_k-\frac{X_k-X_k^{-1}A}{2}=\frac{X_k+X_k^{-1}A}{2}. \end{align*} $$ We take the initial value $X_0=A$ and terminate when $\norm{X_k-X_{k-1}}\lt 10^{-12}$. The algorithm can thus be implemented as follows. 
function mat_sqrt(A)
    """Find the square root of a symmetric positive definite matrix.

    Parameters
    ----------
    A : Matrix{Float64}, etc.
        The matrix to find the square root of.
    
    Returns
    -------
    sqA : Matrix{Float64}
        The square root of the given matrix.
    """
    X = copy(A)
    for _ = 1 : 1000
        X_new = (inv(X) * A + X) / 2
        if norm(X_new - X) < 1e-12
            break
        end
        X = X_new
    end
    return X
end

Solution. We can implement Jacobi's method as follows. 
function _generate_Rpq(n, p, q, phi)
    """Generate R_pq for the Jacobi method."""
    R, c, s = Array{Float64}(I, n, n), cos(phi), sin(phi)
    R[p, p] = c
    R[q, q] = c
    R[p, q] = s
    R[q, p] = -s
    return R
end

function jacobi_eigen(n, A)
    """The Jacobi method for eigenproblem.
    
    Parameters
    ----------
    n : Int64, etc.
        The dimension of the matrix to be found eigenvalues and eigenvectors of.
    A : Matrix{Float64}, etc.
        The matrix to be found eigenvalues of.

    Returns
    -------
    eigvals : Array{Float64}
        The array of eigenvalues of the given matrix.
    eigvecs : Matrix{Float64}
        The matrix where each column is an eigenvector of the given matrix.

    Notes
    -----
    Note that the eigenvalues and the eigenvectors given in `eigvals` and `eigvecs`
    respectively are in the same order.
    """
    D, eigvecs, p, q, target = copy(A), Matrix{Float64}(I, n, n), 0, 0, -1
    for i = 1 : n
        for j = 1 : n
            if i != j && abs(D[i, j]) > target
                p, q, target = i, j, abs(D[i, j])
            end
        end
    end
    for _ = 1 : 1000
        phi = atan(2 * D[p, q] / (D[q, q] - D[p, p])) / 2
        R = _generate_Rpq(n, p, q, phi)
        D = R' * D * R
        eigvecs *= R
        target, sum_of_squares = -1, 0
        for i = 1 : n
            for j = 1 : n
                if i != j
                    sum_of_squares += D[i, j] ^ 2
                    if abs(D[i, j]) > target
                        p, q, target = i, j, abs(D[i, j])
                    end
                end
            end
        end
        if sqrt(sum_of_squares) <= 1e-8
            break
        end
    end
    return diag(D), eigvecs
end

Solution. The QR algorithm for finding singular values can be implemented as follows. 
function qr_svd(n, A)
    """The QR algorithm.

    Parameters
    ----------
    n : Int64, etc.
        The dimension of the matrix to find singular values of.
    A : Matrix{Float64}, etc.
        The matrix to find singular values of.
    
    Returns
    -------
    singvals : Array{Float64}
        The array of singular values of the given matrix.
    """
    B = A' * A
    prev = 1
    for _ = 1 : 1000
        (Q, R) = qr(B)
        B = R * Q
        sum_of_squares = 0
        for i = 2 : n
            for j = 1 : (i - 1)
                sum_of_squares += B[i, j] ^ 2
            end
        end
        if abs((sqrt(sum_of_squares) - prev) / prev) <= 1e-6
            break
        end
        prev = sum_of_squares
    end
    return sqrt.(diag(B))
end

Problem 4.4 - Chebyshev Polynomials

There are two commonly used equivalent formulas defining Chebyshev polynomials, which are $$ \begin{align*} T_n(x)=\cos(n\arccos x) \end{align*} $$ and $$ \begin{align*} T_0(x)={} & 1, \\ T_1(x)={} & x, \\ T_{n+1}(x)={} & 2xT_n(x)-T_{n-1}(x). \end{align*} $$ Show that these formulas are equivalent.
Proof. First we prove that the trigonometric form satisfies the recurrence relation. Clearly, we have that $$ \begin{align*} T_0(x)=\cos0=1,\qquad T_1(x)=\cos(\arccos x)=x. \end{align*} $$ Moreover, we can compute that $$ \begin{align*} T_{n+1}(x)={} & \cos((n+1)\arccos x)=\cos(\arccos x+n\arccos x) \nonumber\\ ={} & \cos(\arccos x)\cos(n\arccos x)-\sin(\arccos x)\sin(n\arccos x) \nonumber\\ ={} & 2\cos(\arccos x)\cos(n\arccos x)-(\sin(\arccos x)\sin(n\arccos x)+\cos(\arccos x)\cos(n\arccos x)) \nonumber\\ ={} & 2\cos(\arccos x)\cos(n\arccos x)-\cos(n\arccos x-\arccos x) \nonumber\\ ={} & 2\cos(\arccos x)\cos(n\arccos x)-\cos((n-1)\arccos x)=2xT_n(x)-T_{n-1}(x). \end{align*} $$ Now we prove the the recurrence relation yields the trigonometric form by strong induction. The base cases $n=0$ and $n=1$ clearly hold. Now we assume that $T_k(x)=\cos(k\arccos x)$ for all $k\leq n$. Then for $k=n+1$, we can use the inductive hypothesis and compute that $$ \begin{align*} T_{n+1}(x)={} & 2xT_n(x)-T_{n-1}(x)=2x\cos(n\arccos x)-\cos((n-1)\arccos x) \nonumber\\ ={} & 2x\cos(n\arccos x)-\cos(n\arccos x-\arccos x) \nonumber\\ ={} & 2x\cos(n\arccos x)-(\cos(n\arccos x)\cos(\arccos x)+\sin(n\arccos x)\sin(\arccos x)) \nonumber\\ ={} & 2x\cos(n\arccos x)-x\cos(n\arccos x)-\sin(n\arccos x)\sin(\arccos x) \nonumber\\ ={} & x\cos(n\arccos x)-\sin(n\arccos x)\sin(\arccos x) \nonumber\\ ={} & \cos(\arccos x)\cos(n\arccos x)-\sin(n\arccos x)\sin(\arccos x) \nonumber\\ ={} & \cos(\arccos x+n\arccos x)=\cos((n+1)\arccos x), \end{align*} $$ so our inductive assumption also holds for $k=n+1$. Therefore, we can conclude that the recurrence relation yields the trigonometric form. We have shown the equivalence between the two forms, so the proof is complete.

function legendre(n, x)
    """Evaluate the Legendre polynomial of degree n at x.

    Parameters
    ----------
    n : Int64, etc.
        The degree of the Legendre polynomial to evalute.
    x : Float64, etc.
        The point to evaluate the Legendre polynomial at.

    Returns
    -------
    val : Float64, etc.
        The Legendre polynomial of degree n at x.
    """
    DP = zeros(Float64, n + 1)
    DP[1] = 1.0
    DP[2] = x
    for i = 2 : n
        DP[i + 1] = ((2 * i - 1) * x * DP[i] - (i - 1) * DP[i - 1]) / i
    end
    return DP[n + 1]
end

function dlegendre(n, x)
    """Evaluate the derivative of the Legendre polynomial of degree n at x.

    Parameters
    ----------
    n : Int64, etc.
        The degree of the Legendre polynomial to evalute.
    x : Float64, etc.
        The point to evaluate the derivative at.

    Returns
    -------
    val : Float64, etc.
        The derivative of the Legendre polynomial of degree n at x.
    """
    DP = zeros(Float64, n + 1)
    DP[1] = 0
    DP[2] = 1
    for i = 2 : n
        DP[i + 1] = ((2 * i - 1) * (legendre(i - 1, x) + x * DP[i]) - (i - 1) * DP[i - 1]) / i
    end
    return DP[n + 1]
end
We then use the Newton's method to find the zeros of the Legendre polynomials. 
function legendre_zeros(n)
    """Find the zeros of the Legendre polynomial of order n.

    The initial guesses would be the Chebyshev nodes, i.e., the zeros
    of the Chebyshev polynomial of the first kind of degree n. They
    are given by
    
        :math:` x_k = cos ( (2k - 1)π / (2n) ), k = 1, ..., n`
    
    The roots of the Chebyshev polynomial of degree n are quite close
    to the roots of the Legendre polynomial of degree n.

    Parameters
    ----------
    n : Int64, etc.
        The order of the Legendre polynomial to find zeros of.
    
    Returns
    -------
    zs : Array{Float64}
        The zeros of the Legendre polynomial of order n. This would be
        an array of length n.
    """
    zs = zeros(Float64, n)
    for i = 1 : n
        root = cos((2 * i - 1) * π / (2 * n))
        while true
            dr = legendre(n, root) / dlegendre(n, root)
            if abs(dr) < 1e-14
                break
            end
            root -= dr
        end
        zs[i] = root
    end
    return zs
end

Since $\phi(h)=\Phi+O(h^p)$, we can write that $$ \begin{align*} \phi(h)-\Phi=C_0h^p+O\bkt{h^{p+1}}. \end{align*} $$ Therefore, we have that $$ \begin{align*} \phi\bkt{\frac{h}{2}}-\Phi=C_0\bkt{\frac{h}{2}}^p+O\bkt{h^{p+1}}. \end{align*} $$ Subtracting $2^p$ times the second equation by the first equation, we can obtain that $$ \begin{align*} 2^p\phi\bkt{\frac{h}{2}}-\phi(h)-\bkt{2^p-1}\Phi=O\bkt{h^{p+1}}\implies\underbrace{\frac{2^p\phi(h/2)-\phi(h)}{2^p-1}}_{=:\Phi^{(1)}(h)}-\Phi=O\bkt{h^{p+1}}. \end{align*} $$ The estimate $\Phi^{(1)}(h)$ has already improved the order of accuracy, and we can further repeat this process. By using the two estimates $\Phi^{(1)}(h)$ (obtained from $\phi(h)$ and $\phi(h/2)$) and $\Phi^{(1)}(h/2)$ (obtained from $\phi(h/2)$ and $\phi(h/4)$), we aim to obtain an estimate of $\Phi$ that is even more accurate then either of these estimates alone. Similar to above, we can write that $$ \begin{align*} \Phi^{(1)}(h)-\Phi=C_1h^{p+1}+O\bkt{h^{p+2}}. \end{align*} $$ Therefore, we also have that $$ \begin{align*} \Phi^{(1)}\bkt{\frac{h}{2}}-\Phi=C_1\bkt{\frac{h}{2}}^{p+1}+O\bkt{h^{p+2}}. \end{align*} $$ Subtracting $2^{p+1}$ times the second equation by the first equation, we can obtain that $$ \begin{align*} 2^{p+1}\Phi^{(1)}\bkt{\frac{h}{2}}-\Phi^{(1)}(h)-\bkt{2^{p+1}-1}\Phi=O\bkt{h^{p+2}}\implies\underbrace{\frac{2^{p+1}\Phi^{(1)}(h/2)-\Phi^{(1)}(h)}{2^{p+1}-1}}_{=:\Phi^{(2)}(h)}-\Phi=O\bkt{h^{p+2}}. \end{align*} $$ The estimate $\Phi^{(2)}(h)$ again improves the order of accuracy. Iterating this process as many times as we want until there is no more $\phi$ to use, we will be able to obtain better estimates of $\Phi$ despite not knowing the value of $p$. Note that in each step we need to approximate $p$. For instance, in the first step, we can estimate that $$ \begin{align*} \frac{\phi(h)-\phi(h/2)}{\phi(h/2)-\phi(h/4)}=2^p, \end{align*} $$ etc., details will not be elaborated here, but in the second step we should use $\Phi^{(1)}$ for approximation and so on.

Problem 6.2 - Discrete Orthogonality

For integers $j\neq k$, prove that $$ \begin{align*} \sum_{n=0}^{N-1}\exp\bkt{\frac{-i2\pi jn}{N}}\exp\bkt{\frac{-i2\pi kn}{N}}=0. \end{align*} $$
Proof. Let $\omega_N=\exp\bkt{i2\pi/N}$ be the $N$th root of unity, then we have that $$ \begin{align*} \sum_{n=0}^{N-1}\exp\bkt{\frac{-i2\pi jn}{N}}\exp\bkt{\frac{-i2\pi kn}{N}}=\sum_{n=0}^{N-1}\omega_N^{-jn}\omega_N^{-kn}=\sum_{n=0}^{N-1}\bkt{\omega_N^{-j-k}}^n=\frac{1-\bkt{\omega_N^{-j-k}}^N}{1-\omega_N^{-j-k}}=\frac{1-\bkt{\omega_N^N}^{-j-k}}{1-\omega_N^{-j-k}}=0. \end{align*} $$ Here we have used the fact that $$ \begin{align*} \omega_N^N=\exp\bkt{\frac{i2\pi N}{N}}=\exp\bkt{i2\pi}=\cos(2\pi)+i\sin(2\pi)=1, \end{align*} $$ so the proof is complete.

Problem 6.3 - Spectral Legendre Differentiation

Suppose that a function $f$ is defined via its Legendre expansion $$ \begin{align*} f(x)=\sum_{n=0}^N\alpha_nP_n(x), \end{align*} $$ where the Legendre polynomials are defined via their three-term recurrence $$ \begin{align*} P_0(x)={} & 1, \\ P_1(x)={} & x, \\ P_{n+1}(x)={} & \frac{2n+1}{n+1}xP_n(x)-\frac{n}{n+1}P_{n-1}(x). \end{align*} $$ The function $f$ can be differentiated spectrally by differentiating term by term, such that $$ \begin{align*} f'(x)=\sum_{n=0}^N\alpha_nP_n'(x). \end{align*} $$ However, $f'$ is just another polynomial, so it can be written in terms of a Legendre expansion as well, such that $$ \begin{align*} f'(x)=\sum_{n=0}^{N-1}\beta_nP_n(x). \end{align*} $$ The coefficients $\beta_n$ can be obtained from the $\alpha_n$ coefficients using the following relationship for Legendre polynomials, such that $$ \begin{align*} P_{n+1}'(x)=(2n+1)P_n(x)+(2(n-2)+1)P_{n-2}(x)+(2(n-4)+1)P_{n-4}(x)+\cdots \end{align*} $$ You can assume that $P_n=0$ for $n\lt 0$. Notice that the Legendre expansion for $f$ contains $N+1$ terms and the expansion for $f'$ contains $N$ terms (since the degree of $f'$ is one less than the degree of $f$). This means that the linear spectral differentiation operator that maps the $\alpha_n$'s to the $\beta_n$'s is an $N\times(N+1)$ matrix.
(a) Write a function in Julia whose input is $N$ an whose output is an $N\times(N+1)$ matrix $D_N$ that converts the $\alpha_n$'s into the $\beta_n$'s. Turn in a printout of this matrix for $N=6$, and make sure that your output is cleanly formatted. Print each entry in scientific notation with $5$ significant digits of accuracy, e.g., 3.1416E+00.
Solution. We can compute that $$ \begin{align*} f'(x)={} & \sum_{n=0}^N\alpha_nP_n'(x) \nonumber\\ ={} & \sum_{n=0}^N\alpha_n(2n-1)P_{n-1}(x)+\sum_{n=0}^N\alpha_n(2n-5)P_{n-3}(x)+\sum_{n=0}^N\alpha_n(2n-9)P_{n-5}(x)+\cdots \nonumber\\ ={} & P_0(x)\bkt{\alpha_1+\alpha_3+\alpha_5+\cdots}+P_1(x)\bkt{3\alpha_2+3\alpha_4+3\alpha_6+\cdots}+P_2(x)\bkt{5\alpha_3+5\alpha_5+\cdots}+\cdots \end{align*} $$ On the other hand, we have that $$ \begin{align*} f'(x)=\sum_{n=0}^{N-1}\beta_nP_n(x)=P_0(x)\beta_0+P_1(x)\beta_1+P_2(x)\beta_2+\cdots \end{align*} $$ Therefore, we must require that $$ \begin{align*} \beta_0=\alpha_1+\alpha_3+\alpha_5+\cdots,\qquad\beta_1=3(\alpha_2+\alpha_4+\alpha_6+\cdots),\qquad\beta_2=5(\alpha_3+\alpha_5+\cdots),\qquad\cdots \end{align*} $$ We then construct the spectral differentiation matrix based on this observation. 
function spectral_diff_mat(N)
    """The matrix representing the spectral differentiation operator.

    A function f can be differentiated spectrally by differentiating term by term,
    such that
        f'(x) = sum_{n=0}^N alpha_n P_n'(x).

    However, f' is just another polynomial, so it can be written in terms of a Legendre
    expansion as well, such that
        f'(x) = sum_{n=0}^{N-1} beta_n P_n(x).
    
    The coefficients beta_n can be obtained from the alpha_n coefficients using the
    following relationship for Legendre polynomials, such that
        P_{n+1}'(x) = (2n+1) P_n(x) + (2(n-2)+1) P_{n-2}(x) + (2(n-4)+1) P_{n-4}(x) + ...
    
    The linear spectral differentiation operator maps the alpha_n's to the beta_n's,
    which is an N*(N+1) matrix.

    Parameters
    ----------
    N : Int64, etc.
        The degree of the Legendre expansion.
    
    Returns
    -------
    D : Matrix{Float64}
        The spectral differentiation matrix.
    """
    D = zeros(Float64, N, N + 1)
    for i = 1 : N
        for j = (i + 1) : 2 : (N + 1)
            D[i, j] = 2 * i - 1
        end
    end
    return D
end

function legendres(n, x)
    """Evaluate the Legendre polynomial of degree up to n at x.

    Parameters
    ----------
    n : Int64, etc.
        The degree of the Legendre polynomial to evalute.
    x : Float64, etc.
        The point to evaluate the Legendre polynomial at.

    Returns
    -------
    val : Vector{Float64}, etc.
        The Legendre polynomials of degrees from 0 to n at x.
    """
    DP = zeros(Float64, n + 1)
    DP[1] = 1.0
    DP[2] = x
    for i = 2 : n
        DP[i + 1] = ((2 * i - 1) * x * DP[i] - (i - 1) * DP[i - 1]) / i
    end
    return DP
end

Problem 6.4 - Acyclic Convolution

For two sequences $f_0,\cdots,f_{m-1}$ and $g_0,\cdots,g_{n-1}$, where $m$ and $n$ are integers, the \textit{acyclic convolution} $h=f*g$ is defined as \begin{align*} h_k=\sum_{j=-\infty}^{\infty}f_{k-j}g_j, \end{align*} where $f_l=0$ if $l\lt 0$ or $l\geq m$, and $g_l=0$ if $l\lt 0$ or $l\geq n$.
(a) For what values of $k$ can you assume that $h_k$ is zero?
Solution. Since $f_l=0$ if $l\lt 0$ or $k\geq m$, and $g_l=0$ if $l\lt 0$ or $l\geq n$, we can reduce the sum to $$ \begin{align*} h_k=\sum_{j=-\infty}^{\infty}f_{k-j}g_j=\sum_{j=0}^{n-1}f_{k-j}g_j=\sum_{j=\max(k+1-m,0)}^{\min(k,n-1)}f_{k-j}g_j. \end{align*} $$ Therefore, if the lower limit of the sum is greater than the upper limit, then the sum can be guaranteed as zero. This means that $\max(k+1-m,0)>\min(k,n-1)$, which solves to $k\lt 0$ or $k>m+n-2$.
(b) How can one use the DFT (and therefore the FFT) to compute all the $h_k$'s?
Solution. We first pad the sequences of $f$ and $g$ with zeros to length $m+n-1$ so that they have the same length. This is feasible since $h_k=0$ for $k\lt 0$ or $k>m+n-2$, as we have shown in the previous part. Then, we can compute the DFT of $f$ and $g$ respectively using FFT. Then, multiply the resulting DFTs $F$ and $G$ elementwise, and compute the IDFT of the result using again FFT. The resulting sequence of length $m+n-1$ will be the acyclic convolution $h$, only including $h_k$ for $0\leq k\leq m+n-2$.
(c) Let $m=128$ and $n=256$. Define the $m$-term sequence $f$ by $$ \begin{align*} f_j= \begin{cases} 1/32, & j=0,\cdots,31, \\ 0, & j=32,\cdots,127, \end{cases} \end{align*} $$ and the $n$-term sequence $g$ by $$ \begin{align*} g_j= \begin{cases} 0, & j=0,\cdots,191, \\ 1/64, & j=192,\cdots,255. \end{cases} \end{align*} $$ Write a code to compute the acyclic convolution $h$ of these two sequences. Include a plot of the sequence $h_k$.
Solution. We implement the strategy mentioned in (b) for computing the acyclic convolution. 
function acyclic_convolution(m, f, n, g)
    """Computes the acyclic convolution using FFT.

    Parameters
    ----------
    m : Int64, etc.
        The sequence f is assumed to be zero for index < 0 or >= m.
        Must be a positive integer.
    f : Vector{Float64}, etc.
        The input sequence f of length m for computing the convolution.
    n : Int64, etc.
        The sequence g is assumed to be zero for index < 0 or >= m.
        Must be a positive integer.
    g : Vector{Float64}, etc.
        The input sequence g of length n for computing the convolution.

    Returns
    -------
    h : Vector{Float64}
        The acyclic convolution of f and g, of length m+n-1.
    """
    F = fft([f; zeros(n - 1)])
    G = fft([g; zeros(m - 1)])
    H = F .* G
    return ifft(H)
end

Problem 6.5 - Fourier Transform

As mentioned briefly in class, for a function $f\in L^2(\RR)$, the Fourier transform is defined as $$ \begin{align*} \hat{f}(\omega)=\frac{1}{2\pi}\int_{-\infty}^{\infty}f(x)\exp\bkt{-i\omega x}dx. \end{align*} $$ The inverse Fourier transform (i.e., the reconstruction of $f$ from $\hat{f}$) is then given by $$ \begin{align*} f(x)=\int_{-\infty}^{\infty}\hat{f}(\omega)\exp\bkt{i\omega x}d\omega. \end{align*} $$

function trapezoidal(a, b, n)
    h, x0 = (b - a) / n, a
    fs = zeros(Float64, n + 1)
    for i = 1 : (n + 1)
        fs[i] = f(x0)
        x0 += h
    end
    return h * (sum(fs) - f(a) / 2 - f(b) / 2) / (2 * π)
end
